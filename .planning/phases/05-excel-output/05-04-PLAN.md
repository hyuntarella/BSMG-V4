  ```typescript
  if (body?.download) {

      },
    });
  }
  ```

저장 버튼 옆에 "엑셀 다운로드" 버튼을 추가한다.

1. **상태 추가**: `const [downloading, setDownloading] = useState(false)`

2. **다운로드 핸들러**:
```typescript
const handleDownload = useCallback(async () => {
  if (!estimate.id || downloading) return;
  setDownloading(true);
  try {
    const res = await fetch(`/api/estimates/${estimate.id}/generate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ download: true }),
    });
    if (!res.ok) throw new Error('엑셀 생성 실패');
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `견적서_${estimate.mgmt_no ?? estimate.id.slice(0, 8)}.xlsx`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  } catch (err) {
    console.error('다운로드 실패:', err);
  } finally {
    setDownloading(false);
  }
}, [estimate.id, estimate.mgmt_no, downloading]);
```

3. **버튼 UI**: 저장 버튼 바로 왼쪽에 배치. Tailwind 스타일:
```tsx
<button
  onClick={handleDownload}
  disabled={downloading || !estimate.id}
  className="px-4 py-2 bg-green-600 text-white rounded-lg text-sm font-medium disabled:opacity-50 hover:bg-green-700 transition-colors"
>
  {downloading ? '생성 중...' : '엑셀'}
</button>
```

4. 파일당 200줄 제한을 지킨다. handleDownload를 추가해도 200줄 이내여야 한다. 현재 197줄이므로 핸들러를 별도 파일로 분리하거나 간결하게 작성한다.

주의: `useState`와 `useCallback` import 확인. 기존 import에 있으면 추가 불필요.
