```typescript
useWakeWord({
  onToggle: voice.toggleRecording,
  onWakeWord: () => {
    if (estimate.sheets.length === 0) {
      voiceFlow.startFlow()
    } else {
      voice.startRecording()
    }
  },
  onEditMode: () => {
    if (estimate.sheets.length > 0 && !editMode.isEditMode) {
      editMode.enterEditMode()
    }
  },
  enabled: voice.status === 'idle' || voice.status === 'speaking',
})
```

**3. onSttText 콜백에 "그만"/"종료" 감지 추가 (per D-04):**

```typescript
onSttText: (text) => {
  // 수정 모드에서 "그만"/"종료" 감지 → 수정 모드 해제
  if (editMode.isEditMode && /그만|종료|멈춰/.test(text.trim())) {
    editMode.exitEditMode()
    voice.stopRecording()  // 현재 녹음 중지
    addLog('assistant', '수정 모드 종료.')
    voice.playTts('수정 모드를 종료합니다.')
    return  // LLM으로 보내지 않음
  }
  // 기존 voiceFlow 분기
  if (voiceFlowRef.current.isActive) {
    voiceFlowRef.current.processText(text)
  } else {
    addLog('user', text)
  }
}
```

**4. VoiceBar onToggle 수정:**
수정 모드 활성 시 toggleRecording만 사용 (수정 모드 진입은 웨이크워드로만):

```typescript
onToggle={() => {
  if (estimate.sheets.length === 0 && !voiceFlow.isActive) {
    voiceFlow.startFlow()

    voice.toggleRecording()
  } else {
    // 시트 있고 수정 모드 아닌 상태 → 수정 모드 진입
    editMode.enterEditMode()
  }
}}
```
